Core Philosophy (Enhanced)
Koda is a powerful scripting language that feels like JavaScript but compiles to C speed:

Easy to learn - JavaScript-familiar syntax, minimal keywords
Easy to extend - Simple library system with #include
Powerful interop - Auto-wrap any C/C++ library with Kodawrap
Production ready - Compile to native binaries
Zero ceremony - No headers to write, no build config

Koda = Lua's simplicity + JavaScript's syntax + C's speed + effortless C/C++ interop

│   └─ exports: player.create(), player.update()
├─ #include "enemy.koda"  
│   └─ exports: enemy.spawn(), enemy.ai()

Rules:

Files are included once (no duplicate loading)
Circular includes are detected and throw error
Path resolution: relative to current file, or from KODA_PATH
All top-level functions/variables in included file become accessible via module namespace

    CURLcode res = curl_easy_perform(curl, AS_STRING(url));

        return make_object({

Kodawrap Implementation
Technology Stack
Kodawrap Tool
├─ libclang (parse C/C++ headers)
├─ Template engine (generate Koda code)
├─ Type inference (smart conversions)
└─ Documentation extractor
Core Algorithm
python# Simplified Kodawrap logic (Python pseudocode)

def generate_wrapper(header_file):
    # 1. Parse with libclang
    index = clang.Index.create()
    tu = index.parse(header_file)
    
    # 2. Extract declarations
    functions = extract_functions(tu.cursor)
    structs = extract_structs(tu.cursor)
    enums = extract_enums(tu.cursor)
    
    # 3. Generate Koda code
    output = []
    output.append('#native "mylibrary"\n')
    
    for func in functions:
        output.append(generate_function_wrapper(func))
    
    for struct in structs:
        output.append(generate_struct_constructor(struct))
    
    for enum in enums:
        output.append(generate_enum_constants(enum))
    
    return '\n'.join(output)

def generate_function_wrapper(func):
    params = ', '.join(p.name for p in func.params)
    doc = extract_documentation(func)
    
    return f'''
{doc}
func {func.name}({params}) {{
    return _ffi_call("{func.name}", {params});
}}
'''

Example: Wrapping SQLite
1. Generate Wrapper
bashkodawrap sqlite3.h -o sqlite.koda
2. Use in Koda
javascript#include "sqlite.koda"

func main() {
    // Open database
    let db = sqlite.open("users.db");
    
    // Create table
    sqlite.exec(db, "CREATE TABLE users (id INT, name TEXT)");
    
    // Insert data
    let stmt = sqlite.prepare(db, "INSERT INTO users VALUES (?, ?)");
    sqlite.bind_int(stmt, 1, 42);
    sqlite.bind_text(stmt, 2, "Jesse");
    sqlite.step(stmt);
    
    // Query data
    let stmt2 = sqlite.prepare(db, "SELECT * FROM users");
    while (sqlite.step(stmt2) == sqlite.ROW) {
        let id = sqlite.column_int(stmt2, 0);
        let name = sqlite.column_text(stmt2, 1);
        print("User:", id, name);
    }
    
    // Auto-closed by GC
}

Complete Example: Game with Raylib
javascript#include <raylib>

let player = {
    x: 400,
    y: 300,
    speed: 5
};

func update() {
    if (raylib.IsKeyDown(raylib.KEY_RIGHT)) {
        player.x += player.speed;
    }
    if (raylib.IsKeyDown(raylib.KEY_LEFT)) {
        player.x -= player.speed;
    }
    if (raylib.IsKeyDown(raylib.KEY_DOWN)) {
        player.y += player.speed;
    }
    if (raylib.IsKeyDown(raylib.KEY_UP)) {
        player.y -= player.speed;
    }
}

func draw() {
    raylib.BeginDrawing();
    raylib.ClearBackground(raylib.BLACK);
    
    raylib.DrawCircle(player.x, player.y, 20, raylib.RED);
    raylib.DrawText("Move with arrows", 10, 10, 20, raylib.WHITE);
    
    raylib.EndDrawing();
}

func main() {
    raylib.InitWindow(800, 600, "Koda Game");
    raylib.SetTargetFPS(60);
    
    while (!raylib.WindowShouldClose()) {
        update();
        draw();
    }
    
    raylib.CloseWindow();
}

Kodawrap CLI Reference
bash# Generate wrapper
kodawrap <header.h> [options]

Options:
  -o, --output <file>        Output .koda file
  -l, --link <lib>           Library to link against
  -I, --include <path>       Add include path
  -c, --config <file>        Use config file
  --docs                     Extract documentation
  --markdown                 Generate markdown docs
  --exclude <pattern>        Exclude functions matching pattern
  --rename <from:to>         Rename function
  --prefix <prefix>          Add prefix to all functions
  --no-constants             Don't generate enum constants
  --no-structs               Don't generate struct constructors

Examples:
  kodawrap SDL2/SDL.h -o sdl2.koda -l SDL2
  kodawrap raylib.h -o raylib.koda --docs
  kodawrap mylib.h -c kodawrap.toml
